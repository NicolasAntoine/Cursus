import numpy as np
def integrate_Euler(F,x,y,xStop,h):    
    X = []
    Y = []
    X.append(x)
    Y.append(y)
    while abs(xStop-x)>10**-12:
        h = min(h,xStop - x)
        y = y + h*F(x,y)
        x = x + h
        X.append(x)
        Y.append(y)
    return np.array(X),np.array(Y)

def integrate_RK4(F,x,y,xStop,h):
    
    def run_kut4(F,x,y,h):
        K0 = h*F(x,y)
        K1 = h*F(x + h/2.0, y + K0/2.0)
        K2 = h*F(x + h/2.0, y + K1/2.0)
        K3 = h*F(x + h, y + K2)
        return (K0 + 2.0*K1 + 2.0*K2 + K3)/6.0    
    
    X = []
    Y = []
    X.append(x)
    Y.append(y)
    while x < xStop:
        h = min(h,xStop - x)
        y = y + run_kut4(F,x,y,h)
        x = x + h
        X.append(x)
        Y.append(y)
    return np.array(X),np.array(Y)

Question 3 - spring–mass system¶

The movement of a spring-mass system is defined by the following differential equation:

$\Large m{\frac {d^{2}x}{dt^{2}}}+c{\frac {dx}{dt}}+kx=0 $

Where:

  • $x(t)$ is the body position at time $t$
  • $t$ is the time variable
  • $m$ is the mass of the body attached to the spring
  • $c$ is the damping coefficient
  • $k$ is the spring constant,

Can you solve the system for

  • $c = 0.95$
  • $k = 5 $
  • $m = 1.5$
  • $x(0) = 10$
  • $x'(0) = 0$

We are particularly interested in the position of the body at $t=15$ and to know (approximately) for how long the body will have oscillations with an amplitude larger than 0.1

def integrate_RK4_with_stopper(F,x,y,xStop,h, should_stop=None):
    
    def run_kut4(F,x,y,h):
        K0 = h*F(x,y)
        K1 = h*F(x + h/2.0, y + K0/2.0)
        K2 = h*F(x + h/2.0, y + K1/2.0)
        K3 = h*F(x + h, y + K2)
        return (K0 + 2.0*K1 + 2.0*K2 + K3)/6.0    
    
    X = []
    Y = []
    X.append(x)
    Y.append(y)
    while x < xStop:
        h = min(h,xStop - x)
        y = y + run_kut4(F,x,y,h)
        x = x + h
        X.append(x)
        Y.append(y)
        if(should_stop is not None and should_stop(x,y)): break
    return np.array(X),np.array(Y)

    y = y[1]
